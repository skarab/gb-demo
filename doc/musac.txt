void main() {
    NR52_REG = 0x80;
    NR51_REG = 0xFF;
    NR50_REG = 0x77;

    // set module before
    switch_hUGE_module(&Intro, (UINT8)&__bank_Intro);
    // then add ISR, or else it play garbage before the first switch_hUGE_module() call 
    __critical {
        add_VBL(ISR_wrapper);
    }
	
	
	
	
void switch_hUGE_module(const hUGESong_t * module, UINT8 module_bank) {
    __critical {
        hUGE_init_banked(module);
        current_module_bank = module_bank;
    }
}



UINT8 current_module_bank = (UINT8)&__bank_part1;
void ISR_wrapper() {
    static UINT8 __save;
    __save = _current_bank;
    SWITCH_ROM_MBC1(current_module_bank);
    hUGE_dosound();
    SWITCH_ROM_MBC1(__save);
}
